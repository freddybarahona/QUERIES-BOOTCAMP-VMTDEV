			void Multi();

		class Suma : Calculadora {
			public override void Suma(int arg1, int arg2){
				return arg1 + arg2;
			}
		class Resta : Calculadora {
			public override void Suma(int arg1, int arg2){
				return arg1 - arg2;
			}
		class Multi : Calculadora {
			public override void Suma(int arg1, int arg2){
				return arg1 * arg2;
			}
		class Division : Calculadora {
			public override void Suma(int arg1, int arg2){
				return arg1 / arg2;
